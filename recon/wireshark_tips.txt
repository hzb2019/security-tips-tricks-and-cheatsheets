FILTERS

  examples:
    http.server contains "Apache"
    http.host matches "\.(php|html)"
    tcp.port in {80 443 8080}
    upper(http.server) contains "APACHE"
    lower(http.server) contains "apache"
    string(frame.number) matches "[13579]$"
